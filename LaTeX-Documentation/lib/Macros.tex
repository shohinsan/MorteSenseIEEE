\newcommand{\img}[3][]{%
    \begin{figure}
        \centering
            \if\relax\detokenize{#1}\relax
        \includegraphics[width=\linewidth]{datasets/images/#2}
        \else
        \includegraphics[height=\linewidth, angle=#1]{datasets/images/#2}
        \fi
        \caption{#3}
        \label{fig:#2}
    \end{figure}
}

% ----- Colored Brand -----
\definecolor{myColor}{RGB}{64, 0, 0}
\pgfkeys{
    /coloredbrand/.is family, /coloredbrand,
    default/.style = {
        text size = \Large,
        text style = \bfseries,
        text color = myColor,
    },
    text size/.store in = \txtsize,
    text style/.store in = \txtstyle,
    text color/.store in = \txtcolor,
}
\newcommand{\coloredbrand}[2][]{
    \pgfkeys{/coloredbrand, default, #1}
    {\txtsize\txtstyle\textcolor{\txtcolor}{#2}}
}
% Usage: \coloredbrand{Name}
