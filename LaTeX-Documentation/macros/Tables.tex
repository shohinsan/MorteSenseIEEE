\newcommand{\tbl}[3]{
    \begin{table}[h]
        #2 % Combination of
        % external and internal
        \caption{#1}
        \label{tab:#3}
    \end{table}
}

% External borders
\newenvironment{tableplus}{
    \begin{adjustbox}{width=\linewidth}
        \rowcolors{1}{gray!10!white}{white}
        \begin{tabular}{|l|m{3cm}|}
            \hline
            }{
            \hline
        \end{tabular}
    \end{adjustbox}
}

% Internal borders
\newcommand{\point}[2]{
    #1 &
    \begin{itemize}
        \vspace{0.25cm}
        #2
    \end{itemize}
    \tabularnewline\hline
}
